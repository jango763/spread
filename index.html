// Точные URL-адреса для получения JSON-данных напрямую const sources = [ { // Главное API курсов ЦБ РФ (требуется конкретный js-файл) url: 'https://www.cbr-xml-daily.ru/daily_json.js', type: 'cbr' }, { // Официальное и бесплатное зеркало курсов (базовая валюта - RUB) // ИСПРАВЛЕНО: Добавлен точный путь к JSON-эндпоинту API url: 'https://er-api.com', type: 'api' } ]; async function fetchCbrRates() { const statusText = document.getElementById('status-text'); const statusIcon = document.getElementById('status-icon'); if (statusText) statusText.innerText = 'Загрузка актуальных курсов...'; if (statusIcon) statusIcon.style.backgroundColor = '#4299e1'; // Синий (загрузка) for (let source of sources) { try { const response = await fetch(source.url); if (!response.ok) throw new Error('Сбой сети'); const data = await response.json(); if (source.type === 'cbr') { // Структура ЦБ РФ: данные лежат в объекте Valute, курс идет в лоб (1 единица валюты = X рублей) if (!data.Valute || !data.Valute.EUR || !data.Valute.USD) throw new Error('Неполные данные ЦБ'); document.getElementById('rateEurBase').value = data.Valute.EUR.Value.toFixed(2); document.getElementById('rateUsdBase').value = data.Valute.USD.Value.toFixed(2); } else if (source.type === 'api') { // Структура ExchangeRate-API: при базовой RUB курс показывает, сколько единиц валюты дают за 1 рубль. // Чтобы получить стоимость валюты в рублях, нужно разделить 1 на это значение. if (!data.rates || !data.rates.EUR || !data.rates.USD) throw new Error('Неполные данные зеркала'); document.getElementById('rateEurBase').value = (1 / data.rates.EUR).toFixed(2); document.getElementById('rateUsdBase').value = (1 / data.rates.USD).toFixed(2); } if (statusText) statusText.innerText = 'Курсы ЦБ РФ успешно обновлены'; if (statusIcon) statusIcon.style.backgroundColor = '#38a169'; // Зеленый (успех) saveInputs(); runCalculation(); return; // Прерываем цикл, так как данные успешно получены } catch (e) { console.warn(`Ошибка загрузки с ${source.url}:`, e.message); } } // Сюда код дойдет, только если оба источника упали if (statusText) statusText.innerText = 'Не удалось обновить курсы. Введите вручную.'; if (statusIcon) statusIcon.style.backgroundColor = '#e53e3e'; // Красный (ошибка) runCalculation(); }
